import sys
import typing
from PyQt5 import QtCore, QtWidgets
from PyQt5.QtWidgets import QApplication,QWidget,QMainWindow
from PyQt5.QtGui import QPalette,QColor

from MainWindow1 import Ui_MainWindow # oluşturduğumuz sayfayı çağırdık...


class MyApp(QtWidgets.QMainWindow):
    def __init__(self):
        super(MyApp,self).__init__()
        self.ui = Ui_MainWindow() # burada aslında biz qtdesigner ile oluşturduğumuz yapıyı kendi kodumuzaa dökmeye başlarız.
        
        self.ui.setupUi(self)
        
        self.ui.cbsinema.stateChanged.connect(self.show_state)
        self.ui.cbkitapokumak.stateChanged.connect(self.show_state)
        self.ui.cbspor.stateChanged.connect(self.show_state)
        
        self.ui.btnsecilenlerial.clicked.connect(self.getAll)
    
    def getAll(self):
        result = ""
        items = self.ui.centralwidget.findChildren(QtWidgets.QCheckBox)  # centralwidget bizim tüm checkboxlarımızın olduğu yer. findchildren ile bir  wigdet içerisindeki istenenleri alabiliriz. ne anlama geliyor?  istersek checkbox değil o layout içerisinde bulunan diğer yapılardanda alabiliriz.
        
        for item in items:
            if item.isChecked():
                print(item.text())
                result += item.text() + "\n" 
        
        self.ui.lblresult.setText(result) # lbl'ye yazıları göndereceğiz.
        
    
           
    def show_state(self,value):
        cb = self.sender() # metot hangi chechboxu çağırdı görebiliriz.
        
        print(cb.text())
        print(cb.isChecked())
        print(value)
        
        
def app():
    app = QApplication(sys.argv)
    win = MyApp()
    win.show()
    sys.exit(app.exec_())

app()

from MainWindow2 import Ui_MainWindow # oluşturduğumuz sayfayı çağırdık...

class MyApp(QtWidgets.QMainWindow):
    def __init__(self):
        super(MyApp,self).__init__()
        self.ui = Ui_MainWindow() # burada aslında biz qtdesigner ile oluşturduğumuz yapıyı kendi kodumuzaa dökmeye başlarız.
        
        self.ui.setupUi(self)
        
        self.ui.turkiye.setChecked(True) # bu şekil olursa uygulama çalıştığında Türkiye seçili olur.
        self.ui.ilkokul.setChecked(True) # default olarak lise seçilmiş olur...
        
        self.ui.turkiye.toggled.connect(self.onClicked)
        self.ui.azerbaycan.toggled.connect(self.onClicked)
        self.ui.yunanistan.toggled.connect(self.onClicked)
        self.ui.almanya.toggled.connect(self.onClicked)
        
        self.ui.lise.toggled.connect(self.onClicked1)
        self.ui.universite.toggled.connect(self.onClicked1)
        self.ui.yukseklisans.toggled.connect(self.onClicked1)
        self.ui.ilkokul.toggled.connect(self.onClicked1)        
        
        self.ui.btnulke.clicked.connect(self.getselectedCountry)
        self.ui.egitimsecimi.clicked.connect(self.getselectedLesson)
        
        
    def onClicked(self):
        rb = self.sender()
        if rb.isChecked():
            print(rb.text())
            
    def onClicked1(self):
        rb = self.sender()
        if rb.isChecked():
            print(rb.text())
                            
    def getselectedCountry(self):
        items = self.ui.grp1.findChildren(QtWidgets.QRadioButton)
        for rb in items:
            if rb.isChecked():
                self.ui.lblulke.setText("Secilen ulke: "+rb.text())
                       
        
    def getselectedLesson(self):
        items = self.ui.grp1.findChildren(QtWidgets.QRadioButton)
        for rb in items:
            if rb.isChecked():
                self.ui.lblegitim.setText("Egitim seviyesi: "+rb.text())

                       
                        
        
        
def app():
    app = QApplication(sys.argv)
    win = MyApp()
    win.show()
    sys.exit(app.exec_())

app()

import sys
import typing
from PyQt5 import QtCore, QtWidgets
from PyQt5.QtWidgets import QApplication,QWidget,QMainWindow
from PyQt5.QtGui import QPalette,QColor

from MainWindow3 import Ui_MainWindow # oluşturduğumuz sayfayı çağırdık...

class MyApp(QtWidgets.QMainWindow):
    def __init__(self):
        super(MyApp,self).__init__()
        self.ui = Ui_MainWindow() # burada aslında biz qtdesigner ile oluşturduğumuz yapıyı kendi kodumuzaa dökmeye başlarız.
        
        self.ui.setupUi(self)
        self.ui.btnload.clicked.connect(self.loadItems)
        self.ui.btnitem.clicked.connect(self.getitem)
        self.ui.pushButton.clicked.connect(self.clear)
        
        
        self.ui.cbsehirler.currentIndexChanged.connect(self.change) # seçimde değişim olunca yeni gelenin indexi gelecek.
        self.ui.cbsehirler.currentIndexChanged[str].connect(self.changeselect) # seçimde değişim olunca yeni gelenin indexi gelecek.
    
    def clear(self):
        self.ui.cbsehirler.clear() # tüm veriler silinir comboboxtan...
        
        
        
    def changeselect(self,text):
        print(text)
        
         
    
    def change(self,index):
        print(index)
                
    def loadItems(self):
        self.ui.cbsehirler.addItems(["Adana","İzmir","Antalya"]) # add item ile bir tane ekleriz addItems ile birden fazla ama o zaman liste içerisinde yollamak lazım...
         
    def getitem(self):
        print(self.ui.cbsehirler.currentText()) # seçili text gelir.
        print(self.ui.cbsehirler.currentIndex()) # hangi indexte o combobox içerisinde...
        
        count = self.ui.cbsehirler.count() # kaç adet veri var görebiliriz.
        
        for i in range(count):
            print(self.ui.cbsehirler.itemText(i)) # verilen indexteki veriyi getirir combobox içerisindeki..
            
            
            
        
            
        
            
        
        
def app():
    app = QApplication(sys.argv)
    win = MyApp()
    win.show()
    sys.exit(app.exec_())

app()
