import sys
import typing
from PyQt5 import QtCore, QtWidgets
from PyQt5.QtWidgets import QApplication,QWidget,QMainWindow
from PyQt5.QtGui import QPalette,QColor

from MainWindow1 import Ui_MainWindow # oluşturduğumuz sayfayı çağırdık...


class MyApp(QtWidgets.QMainWindow):
    def __init__(self):
        super(MyApp,self).__init__()
        self.ui = Ui_MainWindow() # burada aslında biz qtdesigner ile oluşturduğumuz yapıyı kendi kodumuzaa dökmeye başlarız.
        
        self.ui.setupUi(self)
        
        self.ui.cbsinema.stateChanged.connect(self.show_state)
        self.ui.cbkitapokumak.stateChanged.connect(self.show_state)
        self.ui.cbspor.stateChanged.connect(self.show_state)
        
        self.ui.btnsecilenlerial.clicked.connect(self.getAll)
    
    def getAll(self):
        result = ""
        items = self.ui.centralwidget.findChildren(QtWidgets.QCheckBox)  # centralwidget bizim tüm checkboxlarımızın olduğu yer. findchildren ile bir  wigdet içerisindeki istenenleri alabiliriz. ne anlama geliyor?  istersek checkbox değil o layout içerisinde bulunan diğer yapılardanda alabiliriz.
        
        for item in items:
            if item.isChecked():
                print(item.text())
                result += item.text() + "\n" 
        
        self.ui.lblresult.setText(result) # lbl'ye yazıları göndereceğiz.
        
    
           
    def show_state(self,value):
        cb = self.sender() # metot hangi chechboxu çağırdı görebiliriz.
        
        print(cb.text())
        print(cb.isChecked())
        print(value)
        
        
def app():
    app = QApplication(sys.argv)
    win = MyApp()
    win.show()
    sys.exit(app.exec_())

app()

    def __init__(self):
        super(MyApp,self).__init__() # aslında bir döngü oluşturdum...

        self.ui.pushButton.clicked.connect(self.totalcal)
    
    
    def totalcal(self):
        start = self.ui.dateEdit_2.date() # tarih bilgisini aldık...
        end = self.ui.dateEdit.date() # tarih bilgisini aldık...
        print(start)
        print(end)
        
        print("Days in month: {0}".format(start.daysInMonth()))
        print("Days in year: {0}".format(start.daysInYear()))
        
        now = QDateTime.currentDateTime() # şu anki zaman bilgisi
        now1 = QDate.currentDate() # şu anki tarih bilgisi.
        now2 = QTime.currentTime() # şu anki saat bilgisi.
        print(str(now) + " " +str(now1)+ " " + str(now2))
         
        
        print("Days: {0}".format(start.daysTo(end))) # aradaki gün farkını görebiliriz.

import sys
import typing
from PyQt5 import QtCore, QtWidgets
from PyQt5.QtWidgets import QApplication,QWidget,QMainWindow
from PyQt5.QtGui import QPalette,QColor
from PyQt5.QtWidgets import QMessageBox,QLineEdit # mesaj kutusu için import ettik mesaj verecek bize...
from PyQt5.QtGui import QIcon # ikon eklemek için kullanırız.
from PyQt5.QtCore import QDate,QTime,QDateTime # tarih,zaman,tarih zaman kullanmak için çağırdık...
from PyQt5.QtWidgets import QInputDialog # bir input oluşturacak add dediğimiz zaman...

from MainWindow6 import Ui_MainWindow # oluşturduğumuz sayfayı çağırdık...


class MyApp(QtWidgets.QMainWindow):
    def __init__(self):
        super(MyApp,self).__init__() # aslında bir döngü oluşturdum...
        self.ui = Ui_MainWindow() # burada aslında biz qtdesigner ile oluşturduğumuz yapıyı kendi kodumuzaa dökmeye başlarız.     
        self.ui.setupUi(self)
        # load students...
        self.loadStudents()
        
        # add new students
        self.ui.btnadd.clicked.connect(self.addstudent)
        
        # edit student
        self.ui.btnedit.clicked.connect(self.editstudent)
        
        # delete student
        self.ui.btnremove.clicked.connect(self.deletestudent)
        
        # up
        
        self.ui.btnup.clicked.connect(self.up)
        
        
        # down
        self.ui.btndown.clicked.connect(self.down)
        
        # sort
        
        self.ui.btnsort.clicked.connect(self.sortStudents)
        
        # close
        
        self.ui.btnexit.clicked.connect(self.close)
        
        
    def down(self):
        currentindex = self.ui.listitems.currentRow() # seçili rowun index numarası. 
        if currentindex != self.ui.listitems.count() - 1:
            item = self.ui.listitems.takeItem(currentindex) #
            self.ui.listitems.insertItem(currentindex + 1,item)
            self.ui.listitems.setCurrentItem(item)
            
            
    def up(self):
        currentindex = self.ui.listitems.currentRow() # seçili rowun index numarası. 
        if currentindex != 0:
            item = self.ui.listitems.takeItem(currentindex) #
            self.ui.listitems.insertItem(currentindex - 1,item)
            self.ui.listitems.setCurrentItem(item)
    
    def sortStudents(self):
        self.ui.listitems.sortItems()        
            
    def deletestudent(self):
        currentindex = self.ui.listitems.currentRow() # seçili rowun index numarası. 
        item = self.ui.listitems.item(currentindex) # elemanı alacağız...
        
        if item is None:
            return #fonksiyonu bitirmek için boş return kullandık...
        
        q = QMessageBox.question(self,"Remove Student","Do you want to remove student? " + str(item.text()), QMessageBox.Yes | QMessageBox.No)
        if q == QMessageBox.Yes:
            item = self.ui.listitems.takeItem(currentindex)
            del item
            
    def close(self):
        quit() # ilgili programı kapatacak...                    
        
    def editstudent(self):
        currentindex = self.ui.listitems.currentRow() # seçili rowun index numarası. 
        item = self.ui.listitems.item(currentindex) # elemanı alacağız...
        
        if item is not None:
            text,ok = QInputDialog.getText(self,"Edit Student","New Name: ",QLineEdit.Normal,item.text())
            if (ok and text is not None) and (text != ""):
                item.setText(text) # isim değiştirme oldu...
                
        
      
    def addstudent(self):
        currentindex = self.ui.listitems.currentRow() # seçili rowun index numarası. 
        text, ok = QInputDialog.getText(self,"New Student","Student Name: ") # getInteger dersek sayı alır... getItem ile bir item alır vsss.
                                           # başlık ismi ve alınacak isim ile ilgili bilgilendirme...        
        
        if (ok and text is not None) and (text != ""):  # ok ve text None değilse. demek.
            self.ui.listitems.insertItem(currentindex,text) # index ve text eklemesi...
        
        
    def loadStudents(self):
        self.ui.listitems.addItems(["ahmet","ali","cinar"]) # birden fazla veri ekletebiliriz. ama liste içerisinde tek eklemek için ise addItem()    
        self.ui.listitems.setCurrentRow(1) # birinci indexteki yani ikinci eleman default olara seçili olur...
    
        
        
         
         
        
def app():
    app = QApplication(sys.argv)
    win = MyApp()
    win.show()
    sys.exit(app.exec_())

app()
